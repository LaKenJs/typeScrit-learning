# 编译选项

#### 1. 自动编译TS文件

```
（1） tsc xxx.ts -w

（2）在目录文件下 创建tsconfig.json

// ts编译器配置文件
// 内容如下，就算没有内容，也会自动编译ts文件
{
  "compilerOptions": {
    "module": "commonjs",
    "target": "es5",
    "sourceMap": true
  },
  "exclude": [
    "node_modules"
  ]
}
```

#### 2. ```tsconfig.json``` 配置选项

```
（1）include 用来指定哪些ts文件需要被编译
	{
		"include": [
			"./src/**/*"(路径，**表示任意目录，*表示任意文件)
		]
	}
（2）exclude 除了哪些文件都被编译
	{
		"exclude": [
			"./src/hello/*"(该文件不被编译)
		]
	}
（3）extends 定义被继承的文件
（4）files 指定被编译的文件列表，一般小型的项目经常用到
```

#### 3. compiler Options 选项

```
{
  "compilerOptions": {
    // 'none', 'commonjs', 'amd', 'system', 'umd', 'es6', 'es2015', 'es2020', 'esnext'.
    "module": "commonjs",
    // 'es3', 'es5', 'es6', 'es2015', 'es2016', 'es2017', 'es2018', 'es2019', 'es2020', 'esnext'.
    "target": "es6",
    // 'ES5', 'ES6', 'ES2015', 'ES7', 'ES2016', 'ES2017', 'ES2018', 'ESNext', 'DOM', 'DOM.Iterable',
    // 'WebWorker', 'ScriptHost', 'ES2015.Core', 'ES2015.Collection', 'ES2015.Generator', 'ES2015.Iterable',
    // 'ES2015.Promise', 'ES2015.Proxy', 'ES2015.Reflect', 'ES2015.Symbol', 'ES2015.Symbol.WellKnown',
    // 'ES2016.Array.Include', 'ES2017.object', 'ES2017.Intl', 'ES2017.SharedMemory', 'ES2017.String',
    // 'ES2017.TypedArrays', 'ES2018.Intl', 'ES2018.Promise', 'ES2018.RegExp', 'ESNext.AsyncIterable',
    // 'ESNext.Array', 'ESNext.Intl', 'ESNext.Symbol'. Requires TypeScript version 2.0 or later.
    "lib": [],
    "sourceMap": false,
    // ts文件和js文件分家，将编译好的js放到指定目录中
    "outDir": "./dist",
    // 将编译的js内容，合并到一个js文件中
    // Only 'amd' and 'system' 只有这两种module的类型 才能使用outFile
    // "outFile": "./dist/app.js"

    // 是否对js文件进行编译
    "allowJs": false,

    // 是否检查js文件的语法规范
    "checkJs": false,

    // 是否删除注释，如果编译时不想把注释也编译到js文件可以使用这个选项
    "removeComments": false,

    // 是否不生成编译后的js文件（不常用）
    // "noEmit": false,

    // 当有报错时，不生成编译后的js文件
    "noEmitOnError": true,

    // 语法选项
    // 包括所有的严格选项，建议项目时 打开
    "strict": true,
    // 是否使用严格模式，编译后的js文件，开头会有”use strict“
    "alwaysStrict": false,

    // 不允许出现隐式any的类型出现
    "noImplicitAny": true,
    
    // 不允许出现不明确类型的this出现
    "noImplicitThis": true,
    
    // 严格检查空值,出现空值是否报错
    "strictNullChecks": true
  },
  "exclude": [
    "node_modules"
  ],
  "include": [
    "./src/**/*"
  ]
}

```
